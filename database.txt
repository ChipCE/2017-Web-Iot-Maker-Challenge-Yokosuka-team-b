CREATE TABLE TrainLog (

doorNumber int not null,
inCounter int not null,
outCounter int not null,
people int not null, 
temperature int not null,
humidity int not null,
cabNumber int not null,
station varchar(64) not null,
PRIMARY KEY (id)
